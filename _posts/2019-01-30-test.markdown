     class Test {
        constructor (name) {
            this.name = name;
        }
        
        get getName () {
            return this.name;
        }
        
        set setName (name) {
            this.name = name;
        }
     }
     
     const test = new Test("Bestests");
     console.log(test.getName);
     // "Bestests"
     test.setName = "bestests";
     console.log(test.getName);
     // "bestests"
  </code>
